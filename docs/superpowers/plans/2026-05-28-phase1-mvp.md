# Claude Code Remote Control — Phase 1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the MVP: view Claude Code desktop sessions on Android in real-time (LAN), see messages and tool calls, and send replies from the phone.

**Architecture:** Claude Code plugin (HTTP hooks) → Desktop Agent (Node.js HTTP+WS server on localhost) → Android app (Kotlin+Compose, OkHttp WS client). All communication is local network only in Phase 1.

**Tech Stack:** Node.js + TypeScript + Express + ws (Desktop Agent), Kotlin + Jetpack Compose + OkHttp + Material 3 (Android), JSON config files (Plugin)

import type { Request, Response } from "express";
import type { HookInput, SessionState, WsEvent, PendingApproval } from "./types.js";

// Shared state (populated by server.ts at startup)
export let sessions: Map<string, SessionState> = new Map();
export let pendingApprovals: Map<string, PendingApproval> = new Map();
export let broadcast: (event: WsEvent) => void = () => {};

export function initHooks(
  sessionMap: Map<string, SessionState>,
  approvalMap: Map<string, PendingApproval>,
  broadcastFn: (event: WsEvent) => void
) {
  sessions = sessionMap;
  pendingApprovals = approvalMap;
  broadcast = broadcastFn;
}

function getOrCreateSession(sessionId: string, cwd: string): SessionState {
  let session = sessions.get(sessionId);
  if (!session) {
    session = {
      sessionId,
      cwd,
      messages: [],
      status: "active",
      requestIdCounter: 0,
    };
    sessions.set(sessionId, session);
  }
  return session;
}

function makeRequestId(sessionId: string): string {
  const s = getOrCreateSession(sessionId, "");
  return `${sessionId}-${++s.requestIdCounter}`;
}

export async function handleSessionStart(req: Request, res: Response) {
  const input = req.body as HookInput;
  const session = getOrCreateSession(input.session_id, input.cwd);
  session.status = "active";

  const event: WsEvent = {
    type: "session.start",
    sessionId: input.session_id,
    timestamp: Date.now(),
    data: { cwd: input.cwd },
  };
  session.messages.push(event);
  broadcast(event);
  res.json({});
}

export async function handleUserPrompt(req: Request, res: Response) {
  const input = req.body as HookInput;
  const session = getOrCreateSession(input.session_id, input.cwd);

  if (input.prompt) {
    const event: WsEvent = {
      type: "message.user",
      sessionId: input.session_id,
      timestamp: Date.now(),
      data: { content: input.prompt },
    };
    session.messages.push(event);
    broadcast(event);
  }
  res.json({});
}

export async function handlePreToolUse(req: Request, res: Response) {
  const input = req.body as HookInput;
  const session = getOrCreateSession(input.session_id, input.cwd);
  const requestId = makeRequestId(input.session_id);

  const event: WsEvent = {
    type: "tool.request",
    requestId,
    sessionId: input.session_id,
    timestamp: Date.now(),
    data: {
      toolName: input.tool_name ?? "unknown",
      params: (input.tool_input ?? {}) as Record<string, unknown>,
    },
  };
  session.messages.push(event);
  broadcast(event);

  // Phase 2: add blocking approval logic here
  // For Phase 1, always allow
  res.json({
    hookSpecificOutput: {
      hookEventName: "PreToolUse",
      permissionDecision: "allow",
    },
  });
}

export async function handlePostToolUse(req: Request, res: Response) {
  const input = req.body as HookInput;
  const session = sessions.get(input.session_id);
  // Find the last tool.request for this session to get the requestId
  const lastRequest = session?.messages
    .filter((m): m is WsEvent & { type: "tool.request" } => m.type === "tool.request")
    .at(-1);

  const event: WsEvent = {
    type: "tool.result",
    requestId: lastRequest?.requestId ?? `${input.session_id}-0`,
    sessionId: input.session_id,
    timestamp: Date.now(),
    data: {
      toolName: input.tool_name ?? "unknown",
      output: JSON.stringify(input.tool_input ?? {}),
      success: true,
    },
  };
  session?.messages.push(event);
  broadcast(event);
  res.json({});
}

export async function handlePermissionRequest(req: Request, res: Response) {
  const input = req.body as HookInput;
  const event: WsEvent = {
    type: "tool.permission_request",
    sessionId: input.session_id,
    timestamp: Date.now(),
    data: {
      toolName: input.tool_name ?? "unknown",
      params: (input.tool_input ?? {}) as Record<string, unknown>,
    },
  };
  broadcast(event);
  res.json({});
}

export async function handleStop(req: Request, res: Response) {
  const input = req.body as HookInput;
  const session = sessions.get(input.session_id);
  if (session) {
    session.status = "idle";
  }

  // Stop hook input includes the assistant's response text
  // We extract it from session file watching, but flag idle here
  broadcast({
    type: "notification",
    sessionId: input.session_id,
    timestamp: Date.now(),
    data: { kind: "idle_prompt" },
  });
  res.json({});
}

export async function handleSessionEnd(req: Request, res: Response) {
  const input = req.body as HookInput;
  const session = sessions.get(input.session_id);
  if (session) {
    session.status = "ended";
  }

  const event: WsEvent = {
    type: "session.end",
    sessionId: input.session_id,
    timestamp: Date.now(),
  };
  session?.messages.push(event);
  broadcast(event);
  res.json({});
}

export async function handleNotification(req: Request, res: Response) {
  const input = req.body as HookInput;
  const event: WsEvent = {
    type: "notification",
    sessionId: input.session_id,
    timestamp: Date.now(),
    data: {
      kind: (input.notification_type as "permission_prompt" | "idle_prompt") ?? "idle_prompt",
    },
  };
  broadcast(event);
  res.json({});
}

import { WebSocketServer, WebSocket } from "ws";
import type { Server } from "http";
import type { WsEvent, MobileCommand, SessionState, PendingApproval } from "./types.js";
import { handleReply } from "./claude-cli.js";

const clients = new Set<WebSocket>();

export function createWsServer(
  httpServer: Server,
  sessions: Map<string, SessionState>,
  pendingApprovals: Map<string, PendingApproval>
) {
  const wss = new WebSocketServer({ server: httpServer, path: "/ws" });

  wss.on("connection", (ws: WebSocket) => {
    clients.add(ws);
    console.log(`[ws] client connected (${clients.size} total)`);

    // Send sync for all active sessions
    for (const session of sessions.values()) {
      ws.send(JSON.stringify({
        type: "sync",
        sessionId: session.sessionId,
        timestamp: Date.now(),
        data: { messages: session.messages },
      } satisfies WsEvent));
    }

    ws.on("message", (raw) => {
      try {
        const cmd = JSON.parse(raw.toString()) as MobileCommand;
        handleCommand(cmd, sessions, pendingApprovals);
      } catch (err) {
        console.error("[ws] invalid message:", err);
      }
    });

    ws.on("close", () => {
      clients.delete(ws);
      console.log(`[ws] client disconnected (${clients.size} total)`);
    });

    ws.on("error", (err) => {
      console.error("[ws] error:", err.message);
      clients.delete(ws);
    });
  });

  console.log("[ws] WebSocket server ready on /ws");
  return wss;
}

export function broadcast(event: WsEvent) {
  const payload = JSON.stringify(event);
  for (const client of clients) {
    if (client.readyState === WebSocket.OPEN) {
      client.send(payload);
    }
  }
}

async function handleCommand(
  cmd: MobileCommand,
  sessions: Map<string, SessionState>,
  pendingApprovals: Map<string, PendingApproval>
) {
  switch (cmd.type) {
    case "reply": {
      const result = await handleReply(cmd);
      console.log(`[ws] reply result: ${result.substring(0, 100)}`);
      break;
    }
    case "approve": {
      const approval = pendingApprovals.get(cmd.requestId);
      if (approval) {
        clearTimeout(approval.timer);
        approval.resolve("allow");
        pendingApprovals.delete(cmd.requestId);
      }
      break;
    }
    case "deny": {
      const approval = pendingApprovals.get(cmd.requestId);
      if (approval) {
        clearTimeout(approval.timer);
        approval.resolve("deny");
        pendingApprovals.delete(cmd.requestId);
      }
      break;
    }
    case "status": {
      // Reply with current session states
      for (const [id, session] of sessions) {
        broadcast({
          type: "sync",
          sessionId: id,
          timestamp: Date.now(),
          data: { messages: session.messages },
        });
      }
      break;
    }
  }
}

import { spawn } from "child_process";
import type { ReplyCommand } from "./types.js";

export function handleReply(cmd: ReplyCommand): Promise<string> {
  return new Promise((resolve, reject) => {
    const child = spawn("claude", [
      "-p",
      cmd.data.content,
      "--continue",
      "--output-format",
      "stream-json",
    ], {
      stdio: ["pipe", "pipe", "pipe"],
    });

    let output = "";

    child.stdout.on("data", (chunk: Buffer) => {
      output += chunk.toString();
      // In a full implementation, we'd parse stream-json lines
      // and broadcast assistant messages in real-time
    });

    child.stderr.on("data", (chunk: Buffer) => {
      console.error("[claude-cli] stderr:", chunk.toString());
    });

    child.on("close", (code) => {
      if (code === 0) {
        resolve(output);
      } else {
        reject(new Error(`claude exited with code ${code}`));
      }
    });

    child.on("error", (err) => {
      reject(err);
    });
  });
}

```typescript
import chokidar from "chokidar";
import fs from "fs/promises";
import path from "path";
import os from "os";
import type { SessionState, WsEvent } from "./types.js";

const SESSIONS_DIR = path.join(os.homedir(), ".claude", "sessions");

export function startSessionWatcher(
  sessions: Map<string, SessionState>,
  broadcast: (event: WsEvent) => void
) {
  console.log(`[sessions] watching ${SESSIONS_DIR}`);

  const watcher = chokidar.watch(SESSIONS_DIR, {
    ignored: /(^|[/\\])\../, // ignore dotfiles
    persistent: true,
    ignoreInitial: true,
  });

  watcher.on("add", (filePath) => {
    parseAndEmit(filePath, sessions, broadcast);
  });

  watcher.on("change", (filePath) => {
    parseAndEmit(filePath, sessions, broadcast);
  });

  return watcher;
}

async function parseAndEmit(
  filePath: string,
  sessions: Map<string, SessionState>,
  broadcast: (event: WsEvent) => void
) {
  try {
    const content = await fs.readFile(filePath, "utf-8");
    // Session files are JSONL — one JSON object per line
    const lines = content.trim().split("\n");

    // Extract session ID from filename
    const sessionId = path.basename(filePath, ".jsonl");

    for (const line of lines) {
      if (!line.trim()) continue;
      try {
        const entry = JSON.parse(line);
        // Check if this line is already processed (simple dedup)
        // For MVP, parse assistant messages from the transcript
        if (entry.type === "assistant" && entry.message?.content) {
          const textBlocks = Array.isArray(entry.message.content)
            ? entry.message.content
                .filter((b: { type: string }) => b.type === "text")
                .map((b: { text: string }) => b.text)
            : [String(entry.message.content)];

          const fullText = textBlocks.join("\n");
          if (fullText.trim()) {
            broadcast({
              type: "message.assistant",
              sessionId,
              timestamp: Date.now(),
              data: { content: fullText },
            });
          }
        }
      } catch {
        // skip malformed lines
      }
    }
  } catch (err) {
    // File might not exist yet or be locked — ignore
  }
}
```

object EventParser {
    private val json = Json { ignoreUnknownKeys = true }

    fun parse(raw: String): SessionEvent? {
        return try {
            val root = json.parseToJsonElement(raw).jsonObject
            val type = root["type"]?.jsonPrimitive?.content ?: return null

            when (type) {
                "session.start" -> json.decodeFromString<SessionStartEvent>(raw)
                "message.user" -> json.decodeFromString<UserMessageEvent>(raw)
                "message.assistant" -> json.decodeFromString<AssistantMessageEvent>(raw)
                "tool.request" -> json.decodeFromString<ToolRequestEvent>(raw)
                "tool.result" -> json.decodeFromString<ToolResultEvent>(raw)
                "tool.permission_request" -> json.decodeFromString<PermissionRequestEvent>(raw)
                "notification" -> json.decodeFromString<NotificationEvent>(raw)
                "session.end" -> json.decodeFromString<SessionEndEvent>(raw)
                "sync" -> json.decodeFromString<SyncEvent>(raw)
                else -> null
            }
        } catch (e: Exception) {

package com.clinotify.data.ws

import com.clinotify.data.model.SessionEvent
import kotlinx.coroutines.channels.Channel
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.receiveAsFlow
import okhttp3.*

class WebSocketClient(private val serverUrl: String) {
    private val client = OkHttpClient.Builder()
        .retryOnConnectionFailure(true)
        .build()

    private var ws: WebSocket? = null
    private val _events = Channel<SessionEvent>(Channel.BUFFERED)
    val events: Flow<SessionEvent> = _events.receiveAsFlow()
    private val _connected = Channel<Boolean>(Channel.CONFLATED)
    val connected: Flow<Boolean> = _connected.receiveAsFlow()

    fun connect() {
        val request = Request.Builder()
            .url(serverUrl)
            .build()

        ws = client.newWebSocket(request, object : WebSocketListener() {
            override fun onOpen(webSocket: WebSocket, response: Response) {
                _connected.trySend(true)
            }

            override fun onMessage(webSocket: WebSocket, text: String) {
                val event = EventParser.parse(text)
                if (event != null) {
                    _events.trySend(event)
                }
            }

            override fun onFailure(webSocket: WebSocket, t: Throwable, response: Response?) {
                _connected.trySend(false)
            }

            override fun onClosed(webSocket: WebSocket, code: Int, reason: String) {
                _connected.trySend(false)
            }
        })
    }

    fun send(command: String) {
        ws?.send(command)
    }

    fun disconnect() {
        ws?.close(1000, "user disconnect")
    }
}

package com.clinotify.viewmodel

import androidx.lifecycle.ViewModel
import androidx.lifecycle.ViewModelProvider
import androidx.lifecycle.viewModelScope
import com.clinotify.data.model.*
import com.clinotify.data.ws.WebSocketClient
import kotlinx.coroutines.flow.*
import kotlinx.coroutines.launch

sealed interface DetailItem {
    data class UserMsg(val content: String) : DetailItem
    data class AssistantMsg(val content: String) : DetailItem
    data class ToolCall(val requestId: String, val toolName: String, val params: String) : DetailItem
    data class ToolOutput(val requestId: String, val toolName: String, val output: String) : DetailItem
    data class Notification(val kind: String) : DetailItem
}

class SessionDetailViewModel(private val sessionId: String) : ViewModel() {
    private val _items = MutableStateFlow<List<DetailItem>>(emptyList())
    val items: StateFlow<List<DetailItem>> = _items.asStateFlow()
    private val _status = MutableStateFlow("")
    val status: StateFlow<String> = _status.asStateFlow()

    fun observeEvents(client: WebSocketClient) {
        viewModelScope.launch {
            client.events.collect { event ->
                if (event.sessionId != sessionId) return@collect

                when (event) {
                    is SyncEvent -> {
                        // Parse synced messages for initial load
                        val parsed = event.data.messages.mapNotNull { el ->
                            parseEvent(el.toString())
                        }
                        _items.value = parsed
                    }
                    is UserMessageEvent -> {
                        _items.update { it + DetailItem.UserMsg(event.data.content) }
                    }
                    is AssistantMessageEvent -> {
                        _items.update { it + DetailItem.AssistantMsg(event.data.content) }
                    }
                    is ToolRequestEvent -> {
                        _items.update {
                            it + DetailItem.ToolCall(
                                event.requestId, event.data.toolName,
                                event.data.params.toString()
                            )
                        }
                    }
                    is ToolResultEvent -> {
                        _items.update {
                            it + DetailItem.ToolOutput(
                                event.requestId, event.data.toolName,
                                event.data.output
                            )
                        }
                    }
                    is SessionEndEvent -> _status.value = "Ended"
                    else -> {}
                }
            }
        }
    }

    fun sendReply(client: WebSocketClient, content: String) {
        val cmd = """{"type":"reply","sessionId":"$sessionId","data":{"content":"$content"}}"""
        client.send(cmd)
    }

    private fun parseEvent(raw: String): DetailItem? = try {
        // Simplified parser for sync items
        when {
            raw.contains("\"message.user\"") -> DetailItem.UserMsg(raw)
            raw.contains("\"message.assistant\"") -> DetailItem.AssistantMsg(raw)
            raw.contains("\"tool.request\"") -> DetailItem.ToolCall("", "", "")
            else -> null
        }
    } catch (e: Exception) { null }
}

class SessionDetailViewModelFactory(
    private val sessionId: String
) : ViewModelProvider.Factory {
    override fun <T : ViewModel> create(modelClass: Class<T>): T {
        @Suppress("UNCHECKED_CAST")
        return SessionDetailViewModel(sessionId) as T
    }
}

package com.clinotify.ui.screens

import androidx.compose.foundation.layout.*
import androidx.compose.foundation.lazy.LazyColumn
import androidx.compose.foundation.lazy.items
import androidx.compose.foundation.lazy.rememberLazyListState
import androidx.compose.material.icons.Icons
import androidx.compose.material.icons.filled.ArrowBack
import androidx.compose.material3.*
import androidx.compose.runtime.*
import androidx.compose.ui.Modifier
import androidx.compose.ui.unit.dp
import androidx.lifecycle.viewmodel.compose.viewModel
import com.clinotify.data.ws.WebSocketClient
import com.clinotify.ui.components.*
import com.clinotify.viewmodel.DetailItem
import com.clinotify.viewmodel.SessionDetailViewModel
import com.clinotify.viewmodel.SessionDetailViewModelFactory

@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun SessionDetailScreen(
    sessionId: String,
    wsClient: WebSocketClient,
    onBack: () -> Unit,
    vm: SessionDetailViewModel = viewModel(
        factory = SessionDetailViewModelFactory(sessionId)
    )
) {
    val items by vm.items.collectAsState()
    val listState = rememberLazyListState()

    LaunchedEffect(Unit) {
        vm.observeEvents(wsClient)
    }

    LaunchedEffect(items.size) {
        if (items.isNotEmpty()) {
            listState.animateScrollToItem(items.size - 1)
        }
    }

    Scaffold(
        topBar = {
            TopAppBar(
                title = { Text(sessionId.take(12)) },
                navigationIcon = {
                    IconButton(onClick = onBack) {
                        Icon(Icons.Default.ArrowBack, "Back")
                    }
                }
            )
        },
        bottomBar = {
            ReplyBar(onSend = { content -> vm.sendReply(wsClient, content) })
        }
    ) { padding ->
        LazyColumn(
            modifier = Modifier
                .fillMaxSize()
                .padding(padding),
            state = listState
        ) {
            items(items) { item ->
                when (item) {
                    is DetailItem.UserMsg -> MessageBubble(item.content, isUser = true)
                    is DetailItem.AssistantMsg -> MessageBubble(item.content, isUser = false)
                    is DetailItem.ToolCall -> ToolCallCard(item.toolName, item.params)
                    is DetailItem.ToolOutput -> ToolOutputCard(item.toolName, item.output)
                    is DetailItem.Notification -> {}
                }
            }
        }
    }
}
